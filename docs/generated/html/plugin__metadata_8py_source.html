259def get_plugin_metadata(plugin_name: str) -> Optional[PluginMetadata]:
260 """Get metadata for a specific plugin."""
261 return PLUGIN_METADATA.get(plugin_name)
262
263
264def get_all_plugin_names() -> List[str]:
265 """Get list of all available plugin names."""
266 return list(PLUGIN_METADATA.keys())
267

269def get_plugins_by_tags(tags: List[str]) -> List[str]:
270 """Get plugins that match any of the specified tags."""
271 matching_plugins = []
272 for name, metadata in PLUGIN_METADATA.items():
273 if any(tag in metadata.profile_tags for tag in tags):
274 matching_plugins.append(name)
275 return matching_plugins

278def get_platform_compatible_plugins() -> List[str]:
279 """Get plugins compatible with the current platform."""
280 current_platform = platform.system().lower()
281 platform_map = {
282 "windows": "windows",
283 "linux": "linux",
284 "darwin": "macos"
285 }
286
287 mapped_platform = platform_map.get(current_platform, current_platform)
288
289 compatible_plugins = []
290 for name, metadata in PLUGIN_METADATA.items():
291 if mapped_platform in metadata.platforms:
292 compatible_plugins.append(name)
293
294 return compatible_plugins

297def get_gpu_dependent_plugins() -> List[str]:
298 """Get plugins that require GPU support."""
299 return [name for name, metadata in PLUGIN_METADATA.items() if metadata.gpu_required]

302def get_core_plugins() -> List[str]:
303 """Get core plugins that are typically always included."""
304 return [name for name, metadata in PLUGIN_METADATA.items() if not metadata.optional]
