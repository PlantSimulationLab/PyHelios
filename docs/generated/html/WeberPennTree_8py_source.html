21@contextmanager
22def _weberpenntree_working_directory():
23 """
24 Context manager that temporarily changes working directory to where WeberPennTree assets are located.
25
26 WeberPennTree C++ code uses hardcoded relative paths like "plugins/weberpenntree/xml/WeberPennTreeLibrary.xml"
27 expecting assets relative to working directory. This manager temporarily changes to the build directory
28 where assets are actually located.
29
30 Raises:
31 RuntimeError: If build directory or WeberPennTree assets are not found, indicating a build system error.
32 """
33 # Find the build directory containing WeberPennTree assets
34 current_dir = Path(__file__).parent
35 repo_root = current_dir.parent
36 build_lib_dir = repo_root / 'pyhelios_build' / 'build' / 'lib'
37 working_dir = build_lib_dir.parent
38 weberpenntree_assets = working_dir / 'plugins' / 'weberpenntree'
39
40 # Validate that build directory and assets exist - fail fast if not
41 if not build_lib_dir.exists():
42 raise RuntimeError(
43 f"PyHelios build directory not found: {build_lib_dir}. "
44 f"WeberPennTree requires native libraries to be built. "
45 f"Run: python build_scripts/build_helios.py --plugins weberpenntree"
46 )
47
48 if not weberpenntree_assets.exists():
49 raise RuntimeError(
50 f"WeberPennTree assets not found: {weberpenntree_assets}. "
51 f"Build system failed to copy WeberPennTree assets. "
52 f"Run: python build_scripts/build_helios.py --clean --plugins weberpenntree"
53 )
54
55 xml_file = weberpenntree_assets / 'xml' / 'WeberPennTreeLibrary.xml'
56 if not xml_file.exists():
57 raise RuntimeError(
58 f"WeberPennTree XML library not found: {xml_file}. "
59 f"Critical WeberPennTree asset missing from build. "
60 f"Run: python build_scripts/build_helios.py --clean --plugins weberpenntree"
61 )
62
63 # Change to build directory where assets are located
64 original_cwd = Path.cwd()
65 try:
66 os.chdir(working_dir)
67 yield
68 finally:
69 os.chdir(original_cwd)
