116 def load_config(self, config_file: str):
117 """
118 Load configuration from YAML file.
119
120 Args:
121 config_file: Path to configuration file
122
123 Raises:
124 ConfigurationError: If configuration file cannot be loaded or is invalid
125 """
126 try:
127 with open(config_file, 'r') as f:
128 config_data = yaml.safe_load(f)
129
130 if not isinstance(config_data, dict):
131 raise ConfigurationError("Configuration file must contain a YAML dictionary")
132
133 self._parse_config(config_data)
134 self.config_file = config_file
135
136 except FileNotFoundError:
137 raise ConfigurationError(f"Configuration file not found: {config_file}")
138 except yaml.YAMLError as e:
139 raise ConfigurationError(f"Invalid YAML in configuration file: {e}")
140 except Exception as e:
141 raise ConfigurationError(f"Error loading configuration: {e}")

143 def _parse_config(self, config_data: Dict[str, Any]):
144 """Parse configuration data into typed config objects."""
145
146 # Parse plugin configuration
147 if 'plugins' in config_data:
148 plugin_data = config_data['plugins']
149 self.plugin_config = PluginConfig(
150 selection_mode=plugin_data.get('selection_mode', 'profile'),
151 profile=plugin_data.get('profile', 'standard'),
152 explicit_plugins=plugin_data.get('explicit_plugins', []),
153 excluded_plugins=plugin_data.get('excluded_plugins', []),
154 include_optional=plugin_data.get('include_optional', True),
155 platform_specific=plugin_data.get('platform_specific', {})
156 )
157
158 # Parse build configuration
159 if 'build' in config_data:
160 build_data = config_data['build']
161 self.build_config = BuildConfig(
162 build_type=build_data.get('build_type', 'Release'),
163 cmake_args=build_data.get('cmake_args', []),
164 parallel_jobs=build_data.get('parallel_jobs', 0),
165 verbose=build_data.get('verbose', False)
166 )
167
168 # Parse system configuration
169 if 'system' in config_data:
170 system_data = config_data['system']
171 self.system_config = SystemConfig(
172 force_mock_mode=system_data.get('force_mock_mode', False),
173 environment=system_data.get('environment', {})
174 )
175
176 # Parse logging configuration
177 if 'logging' in config_data:
178 logging_data = config_data['logging']
179 self.logging_config = LoggingConfig(
180 level=logging_data.get('level', 'INFO'),
181 plugin_detection=logging_data.get('plugin_detection', True),
182 build_logging=logging_data.get('build_logging', True)
183 )

185 def resolve_plugin_selection(self) -> List[str]:
186 """
187 Resolve the final plugin selection based on configuration.
188
189 Returns:
190 List of plugin names to build
191
192 Raises:
193 ConfigurationError: If plugin configuration is invalid
194 """
195 plugins = []
196
197 if self.plugin_config.selection_mode == "profile":
198 # Use profile-based selection
199 try:
200 plugins = filter_profile_by_platform(self.plugin_config.profile)
201 except ValueError as e:
202 raise ConfigurationError(f"Invalid profile '{self.plugin_config.profile}': {e}")
203
204 elif self.plugin_config.selection_mode == "explicit":
205 # Use explicit plugin list
206 plugins = list(self.plugin_config.explicit_plugins)
207
208 elif self.plugin_config.selection_mode == "auto":
209 # Auto-select based on system capabilities
210 resolver = PluginDependencyResolver()
211 gpu_available = resolver._check_cuda()
212
213 if gpu_available:
214 plugins = filter_profile_by_platform("gpu-accelerated")
215 else:
216 plugins = filter_profile_by_platform("standard")
217 else:
218 raise ConfigurationError(
219 f"Invalid selection_mode '{self.plugin_config.selection_mode}'. "
220 "Must be 'profile', 'explicit', or 'auto'"
221 )
222
223 # Apply platform-specific modifications
224 current_platform = self._get_current_platform()
225 if current_platform in self.plugin_config.platform_specific:
226 platform_config = self.plugin_config.platform_specific[current_platform]
227
228 # Add platform-specific plugins
229 if 'additional_plugins' in platform_config:
230 plugins.extend(platform_config['additional_plugins'])
231
232 # Remove platform-specific exclusions
233 if 'excluded_plugins' in platform_config:
234 for plugin in platform_config['excluded_plugins']:
235 if plugin in plugins:
236 plugins.remove(plugin)
237
238 # Apply global exclusions
239 for plugin in self.plugin_config.excluded_plugins:
240 if plugin in plugins:
241 plugins.remove(plugin)
242
243 # Remove duplicates and maintain order
244 seen = set()
245 unique_plugins = []
246 for plugin in plugins:
247 if plugin not in seen:
248 seen.add(plugin)
249 unique_plugins.append(plugin)
250
251 return unique_plugins

253 def _get_current_platform(self) -> str:
254 """Get current platform name for configuration matching."""
255 system = platform.system().lower()
256 platform_map = {
257 "windows": "windows",
258 "linux": "linux",
259 "darwin": "macos"
260 }
261 return platform_map.get(system, system)

263 def get_cmake_args(self) -> List[str]:
264 """Get complete CMake arguments including build type."""

279 def validate_configuration(self) -> Dict[str, Any]:
280 """
281 Validate the current configuration.
282
283 Returns:
284 Dictionary with validation results
285 """
286 issues = []
287 warnings = []
288
289 # Validate plugin selection
290 try:
291 plugins = self.resolve_plugin_selection()
292
293 # Check if any plugins were selected
294 if not plugins:
295 issues.append("No plugins selected - this may result in limited functionality")
296
297 # Use dependency resolver to validate
298 resolver = PluginDependencyResolver()
299 validation = resolver.validate_configuration(plugins)
300
301 if validation['invalid_plugins']:
302 issues.extend([f"Invalid plugin: {p}" for p in validation['invalid_plugins']])
303
304 if validation['platform_incompatible']:
305 warnings.extend([f"Platform incompatible plugin: {p}" for p in validation['platform_incompatible']])
306
307 except ConfigurationError as e:
308 issues.append(str(e))
309
310 # Validate build configuration
311 valid_build_types = ['Release', 'Debug', 'RelWithDebInfo', 'MinSizeRel']
312 if self.build_config.build_type not in valid_build_types:
313 issues.append(f"Invalid build_type '{self.build_config.build_type}'. Must be one of: {valid_build_types}")
314
315 # Validate logging configuration
316 valid_log_levels = ['DEBUG', 'INFO', 'WARNING', 'ERROR', 'CRITICAL']
317 if self.logging_config.level not in valid_log_levels:
318 issues.append(f"Invalid log level '{self.logging_config.level}'. Must be one of: {valid_log_levels}")
319
320 return {
321 'valid': len(issues) == 0,
322 'issues': issues,
323 'warnings': warnings,
324 'config_file': self.config_file
325 }
