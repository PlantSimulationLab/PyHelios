1"""
2Dynamic Asset Path Resolution for PyHelios
3
4This module provides working-directory-independent asset path resolution
5for PyHelios plugins. It automatically locates assets in the helios-core
6source directory and sets up environment variables for C++ plugins.
7
8The key principle is NO MANUAL COPYING - all assets are referenced directly
9from their source locations to ensure they stay synchronized with the
10helios-core repository.
11"""

29 def _find_helios_core(self) -> Optional[Path]:
30 """
31 Find the helios-core directory relative to PyHelios package.
32
33 Returns:
34 Path to helios-core directory, or None if not found
35 """
36 if self._helios_core_path is not None:
37 return self._helios_core_path
38
39 # Start from the PyHelios package directory
40 pyhelios_root = Path(__file__).parent.parent.parent
41
42 # Look for helios-core as a subdirectory
43 helios_core_candidates = [
44 pyhelios_root / 'helios-core',
45 pyhelios_root.parent / 'helios-core', # In case PyHelios is in a subdirectory
46 ]
47
48 for candidate in helios_core_candidates:
49 if candidate.exists() and candidate.is_dir():
50 # Verify it's actually the helios-core by checking for key directories
51 if (candidate / 'core').exists() and (candidate / 'plugins').exists():
52 self._helios_core_path = candidate
53 logger.info(f"Found helios-core at: {candidate}")
54 return candidate
55
56 logger.warning("helios-core directory not found - asset paths may not work correctly")
57 return None

59 def get_helios_core_path(self) -> Optional[Path]:
60 """Get the path to the helios-core directory."""
61 return self._find_helios_core()

149 def set_environment_variables(self) -> None:
150 """
151 Set environment variables for C++ plugins to find their assets.
152
153 This method sets the following environment variables:
154 - HELIOS_ASSET_ROOT: Path to helios-core directory
155 - HELIOS_WEBERPENNTREE_PATH: Path to WeberPennTree assets
156 - HELIOS_VISUALIZER_PATH: Path to Visualizer assets
157 - HELIOS_CANOPYGENERATOR_PATH: Path to CanopyGenerator assets
158 - HELIOS_RADIATION_PATH: Path to Radiation assets
159 """
160 helios_core = self.get_helios_core_path()
161 if helios_core:
162 os.environ['HELIOS_ASSET_ROOT'] = str(helios_core)
163 logger.debug(f"Set HELIOS_ASSET_ROOT = {helios_core}")
164
165 asset_paths = self.get_all_asset_paths()
166
167 for plugin_name, path in asset_paths.items():
168 if path is not None:
169 env_var = f"HELIOS_{plugin_name.upper()}_PATH"
170 os.environ[env_var] = path
171 logger.debug(f"Set {env_var} = {path}")
172 else:
173 logger.warning(f"Could not set asset path for plugin: {plugin_name}")

194 def validate_assets(self) -> Dict[str, Dict[str, Any]]:
195 """
196 Validate that all expected asset directories and files exist.
197
198 Returns:
199 Dictionary with validation results for each plugin
200 """
201 results = {}
202 asset_paths = self.get_all_asset_paths()
203
204 for plugin_name, base_path in asset_paths.items():
205 result = {
206 'base_path': base_path,
207 'exists': False,
208 'subdirectories': {},
209 'files_found': 0
210 }
211
212 if base_path and Path(base_path).exists():
213 result['exists'] = True
214 base_dir = Path(base_path)
215
216 # Check for expected subdirectories based on plugin
217 expected_subdirs = self._get_expected_subdirectories(plugin_name)
218
219 for subdir in expected_subdirs:
220 subdir_path = base_dir / subdir
221 result['subdirectories'][subdir] = {
222 'exists': subdir_path.exists(),
223 'files': list(subdir_path.glob('*')) if subdir_path.exists() else []
224 }
225
226 if subdir_path.exists():
227 result['files_found'] += len(list(subdir_path.glob('*')))
228
229 results[plugin_name] = result
230
231 return results

233 def _get_expected_subdirectories(self, plugin_name: str) -> list:
234 """Get expected subdirectories for a given plugin."""
235 subdirs_map = {
236 'weberpenntree': ['leaves', 'wood', 'xml'],
237 'visualizer': ['shaders', 'textures', 'fonts'],
238 'canopygenerator': ['textures'],
239 'radiation': ['spectral_data']
240 }
241 return subdirs_map.get(plugin_name, [])
