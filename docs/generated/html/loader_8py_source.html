23class MockLibrary:
24 """Mock library that provides no-op implementations of all functions."""
25
26 def __init__(self):
27 self._functions = {}
28 logger.info("Using mock library mode - no native Helios functionality available")
29
30 def __getattr__(self, name: str) -> Callable:
31 """Return a mock function for any requested function."""
32 if name not in self._functions:
33 def mock_function(*args, **kwargs):
34 raise RuntimeError(
35 f"Cannot call '{name}' - PyHelios is running in mock mode. "
36 f"Native Helios library not available on this platform. "
37 f"Install platform-appropriate Helios binaries to enable full functionality."
38 )
39
40 mock_function.__name__ = f"mock_{name}"
41 self._functions[name] = mock_function
42
43 return self._functions[name]

110 def get_library_paths(self) -> Dict[str, str]:
111 """Get all possible library paths for current platform."""
112 if self.platform_name not in self.LIBRARY_CONFIG:
113 return {}

115 config = self.LIBRARY_CONFIG[self.platform_name]

121 # Alternative library paths
122 for i, alt in enumerate(config['alternatives']):
123 paths[f'alt_{i}'] = os.path.join(self.plugin_dir, alt)

125 # Dependency paths

127 paths[f'dep_{i}'] = os.path.join(self.plugin_dir, dep)

131 def check_dependencies(self) -> bool:
132 """Check if optional dependencies are available."""
133 if self.platform_name not in self.LIBRARY_CONFIG:
134 return True
135
136 config = self.LIBRARY_CONFIG[self.platform_name]
137 missing_deps = []
138
139 for dep in config['dependencies']:

141 if not os.path.exists(dep_path):

143

165 def load_library(self, force_mock: bool = False) -> ctypes.CDLL:
166 """
167 Load the appropriate native library for the current platform.
168
169 Args:
170 force_mock: If True, use mock library regardless of availability
171
172 Returns:
173 Loaded library or mock library
174
175 Raises:
176 LibraryLoadError: If library loading fails and mock mode is disabled
177 """
178 # Check for explicit mock mode enabling
179 mock_mode_enabled = (
180 force_mock or
181 os.getenv('PYHELIOS_MOCK_MODE', '').lower() in ('1', 'true', 'yes', 'on') or
182 os.getenv('PYHELIOS_DEV_MODE', '').lower() in ('1', 'true', 'yes', 'on')
183 )
184
185 if mock_mode_enabled:
186 logger.info("Mock mode explicitly enabled")
187 self.library = MockLibrary()
188 self.is_mock = True
189 return self.library
190
191 # Check if platform is supported
192 if self.platform_name not in self.LIBRARY_CONFIG:
193 raise LibraryLoadError(
194 f"Platform '{self.platform_name}' is not officially supported by PyHelios. "
195 f"Supported platforms: {list(self.LIBRARY_CONFIG.keys())}. "
196 f"To enable development mode without native libraries, set PYHELIOS_DEV_MODE=1"
197 )
198
199 config = self.LIBRARY_CONFIG[self.platform_name]
200 paths = self.get_library_paths()
201
202 # Automatically configure library path for Linux/macOS
203 if self.platform_name in ['Linux', 'Darwin']:
204 self._setup_library_path()
205
206 # Check dependencies
207 self.check_dependencies()
208
209 # Try to load primary library
210 primary_path = paths.get('primary')
211 if primary_path and os.path.exists(primary_path):
212 try:
213 self.library = config['loader'](primary_path)
214 logger.info(f"Successfully loaded primary library: {primary_path}")
215 return self.library
216 except Exception as e:
217 logger.warning(f"Failed to load primary library {primary_path}: {e}")
218
219 # Try alternative libraries
220 for key, alt_path in paths.items():
221 if key.startswith('alt_') and os.path.exists(alt_path):
222 try:
223 self.library = config['loader'](alt_path)
224 logger.info(f"Successfully loaded alternative library: {alt_path}")
225 return self.library
226 except Exception as e:
227 logger.warning(f"Failed to load alternative library {alt_path}: {e}")
228
229 # No library could be loaded - raise error
230 available_files = [f for f in os.listdir(self.plugin_dir) if f.endswith(('.dll', '.dylib', '.so'))] if os.path.exists(self.plugin_dir) else []
231
232 error_msg = (
233 f"Failed to load native Helios library for platform '{self.platform_name}'. "
234 f"Expected files: {config['primary']} or {config['alternatives']}. "
235 )
236
237 if available_files:
238 error_msg += f"Available library files: {available_files}. "
239 else:
240 error_msg += "No library files found in plugin directory. "
241
242 error_msg += (
243 f"To fix this issue:\n"
244 f"1. Build native libraries: python build_scripts/build_helios.py\n"
245 f"2. Or enable development mode: set PYHELIOS_DEV_MODE=1\n"
246 f"Plugin directory: {self.plugin_dir}"
247 )
248
249 raise LibraryLoadError(error_msg)

277 def validate_library(self) -> bool:
278 """
279 Validate that the loaded library has expected functions.
280
281 Returns:
282 True if library appears to be valid, False otherwise
283 """
284 if self.is_mock:
285 return True # Mock library is always "valid"
286
287 if not self.library:
288 return False
289
290 # List of core functions that should be available
291 core_functions = [
292 'createContext',
293 'destroyContext',
294 'addPatchWithCenterAndSize',
295 'getPrimitiveCount',
296 'getPrimitiveType',
297 'markGeometryClean',
298 'markGeometryDirty',
299 'isGeometryDirty'
300 ]
301
302 missing_functions = []
303 for func_name in core_functions:
304 try:
305 func = getattr(self.library, func_name)
306 if not callable(func):
307 missing_functions.append(func_name)
308 except AttributeError:
309 missing_functions.append(func_name)
310
311 if missing_functions:
312 logger.error(f"Library validation failed - missing functions: {missing_functions}")
313 return False
314
315 logger.info("Library validation passed - all core functions available")
316 return True

337def _find_library_directory() -> str:
338 """
339 Find the directory containing the built Helios library.
340
341 Returns:
342 Path to directory containing library files
343
344 Raises:
345 LibraryLoadError: If no library directory can be found
346 """
347 # Start from the current file location
348 current_dir = os.path.dirname(os.path.abspath(__file__))
349 pyhelios_root = os.path.dirname(os.path.dirname(current_dir)) # Go up to PyHelios root
350
351 # Expected library location after build
352 expected_location = os.path.join(pyhelios_root, 'pyhelios_build', 'build', 'lib')
353
354 platform_name = platform.system()
355 if platform_name == 'Darwin': # macOS
356 library_names = ['libhelios.dylib', 'libCHelios.dylib', 'CHelios.dylib']
357 elif platform_name == 'Windows':
358 library_names = ['libhelios.dll', 'CHelios.dll', 'helios.dll']
359 elif platform_name == 'Linux':
360 library_names = ['libhelios.so', 'libCHelios.so', 'CHelios.so']
361 else:
362 raise LibraryLoadError(
363 f"Platform '{platform_name}' is not supported by PyHelios. "
364 f"Supported platforms: Windows, macOS, Linux"
365 )
366
367 # Check the expected build location
368 if os.path.exists(expected_location):
369 for lib_name in library_names:
370 lib_path = os.path.join(expected_location, lib_name)
371 if os.path.exists(lib_path):
372 logger.debug(f"Found library at: {lib_path}")
373 return expected_location
374
375 # Library not found - provide clear error with actionable solution
376 raise LibraryLoadError(
377 f"Native Helios library not found in expected location: {expected_location}\n"
378 f"Expected library files: {', '.join(library_names)}\n\n"
379 f"To fix this issue:\n"
380 f"1. Build native libraries: ./build_scripts/build_helios --plugins visualizer\n"
381 f"2. Or enable development mode: set PYHELIOS_DEV_MODE=1\n\n"
382 f"PyHelios follows a fail-fast philosophy - if you expected a native library "
383 f"to be available but it's not, something is wrong and needs to be fixed."
384 )
