Solar Position Plugin

The Solar Position plugin provides accurate solar position calculations and sun angle modeling for radiation and energy balance simulations.

Overview

The Solar Position plugin enables:

  • Accurate solar position calculations using astronomical algorithms
  • Sun angle determination for any location, date, and time
  • Solar radiation geometry for lighting simulations
  • Integration with radiation modeling for realistic sun positioning

Basic Usage

from pyhelios.plugins import SolarPosition
from pyhelios import DataTypes
import datetime
# Create solar position calculator
solar = SolarPosition()
# Set location (latitude, longitude)
solar.set_location(latitude=38.5, longitude=-121.8) # Davis, CA
# Set date and time
date_time = datetime.datetime(2023, 6, 21, 12, 0) # Summer solstice, noon
solar.set_datetime(date_time)
# Get sun position
elevation = solar.get_elevation() # degrees above horizon
azimuth = solar.get_azimuth() # degrees from north
print(f"Sun elevation: {elevation:.2f}°")
print(f"Sun azimuth: {azimuth:.2f}°")

Integration with Radiation Modeling

Setting Up Solar Radiation

from pyhelios import Context, RadiationModel
from pyhelios.plugins import SolarPosition
context = Context()
solar = SolarPosition()
# Set location and time
solar.set_location(latitude=35.0, longitude=-118.0) # California
solar.set_datetime(datetime(2023, 7, 15, 14, 0)) # July 15, 2 PM
# Get sun direction for radiation model
sun_direction = solar.get_sun_direction_helios()
# Use in radiation simulation
try:
with RadiationModel(context) as radiation:
radiation.add_radiation_band("PAR")
# Add solar radiation source
radiation.add_collimated_radiation_source(
band="PAR",
direction=sun_direction,
flux_density=1200 # W/m²
)
radiation.run_band("PAR")
except Exception as e:
print(f"Radiation modeling not available: {e}")

Time Series Analysis

Daily Sun Path

# Calculate sun path for entire day
date = datetime(2023, 6, 21) # Summer solstice
solar.set_datetime(date)
sun_path = []
for hour in range(24):
solar.set_hour_of_day(hour)
elevation = solar.get_elevation()
azimuth = solar.get_azimuth()
# Only record when sun is above horizon
if elevation > 0:
sun_path.append({
'hour': hour,
'elevation': elevation,
'azimuth': azimuth
})
print(f"Sun visible for {len(sun_path)} hours")

Advanced Features

Atmospheric Corrections

# Enable atmospheric refraction correction
solar.enable_atmospheric_refraction(True)
# Set atmospheric conditions
solar.set_atmospheric_pressure(1013.25) # hPa
solar.set_temperature(20.0) # Celsius
solar.set_humidity(0.50) # relative humidity
# Get refraction-corrected elevation
uncorrected = solar.get_elevation(atmospheric_correction=False)
corrected = solar.get_elevation(atmospheric_correction=True)
print(f"Uncorrected elevation: {uncorrected:.3f}°")
print(f"Corrected elevation: {corrected:.3f}°")

Solar Radiation Geometry

# Calculate solar radiation components

extraterrestrial_irradiance = solar.get_extraterrestrial_irradiance() # W/m²
# Estimate direct normal irradiance (simple clear-sky model)
direct_normal = extraterrestrial_irradiance * 0.87**(airmass**0.678)

print(f"Extraterrestrial irradiance: {extraterrestrial_irradiance:.0f} W/m²")
print(f"Estimated direct normal irradiance: {direct_normal:.0f} W/m²")

Performance Tips

# For repeated calculations with same location
solar.set_location(latitude=37.7749, longitude=-122.4194) # San Francisco
# Cache calculation parameters
solar.cache_location_parameters(True)
# Batch calculations for time series
times = [datetime(2023, 6, 21, hour, 0) for hour in range(6, 19)] # 6 AM to 6 PM
sun_positions = solar.calculate_batch(times)
for time, (elevation, azimuth) in zip(times, sun_positions):
print(f"{time.hour}:00 - Elevation: {elevation:.1f}°, Azimuth: {azimuth:.1f}°")
