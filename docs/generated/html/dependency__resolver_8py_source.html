45 def resolve_dependencies(self, requested_plugins: List[str],
46 include_optional: bool = True,
47 strict_mode: bool = False) -> ResolutionResult:
48 """
49 Resolve plugin dependencies and conflicts.
50
51 Args:
52 requested_plugins: List of requested plugin names
53 include_optional: Whether to include optional dependencies
54 strict_mode: If True, fail on any missing dependencies
55
56 Returns:
57 ResolutionResult with final plugin list and any issues
58 """
59 result = ResolutionResult(
60 status=ResolutionStatus.SUCCESS,
61 final_plugins=[],
62 added_plugins=[],
63 removed_plugins=[],
64 warnings=[],
65 errors=[],
66 system_check_results={}
67 )
68
69 # Validate requested plugins exist
70 valid_plugins, invalid_plugins = self._validate_plugins(requested_plugins)
71 if invalid_plugins:
72 result.errors.extend([f"Unknown plugin: {p}" for p in invalid_plugins])
73 if strict_mode:
74 result.status = ResolutionStatus.ERROR
75 return result
76
77 # Filter by platform compatibility
78 platform_compatible = [p for p in valid_plugins if p in self.platform_plugins]
79 platform_incompatible = [p for p in valid_plugins if p not in self.platform_plugins]
80
81 if platform_incompatible:
82 result.removed_plugins.extend(platform_incompatible)
83 result.warnings.extend([
84 f"Plugin '{p}' not supported on {platform.system()}"
85 for p in platform_incompatible
86 ])
87
88 # Add plugin dependencies
89 final_plugins = set(platform_compatible)
90 for plugin in platform_compatible:
91 deps = self._get_plugin_dependencies(plugin)
92 new_deps = [d for d in deps if d not in final_plugins]
93 final_plugins.update(deps)
94 result.added_plugins.extend(new_deps)
95
96 # Validate system dependencies
97 system_results = self._check_system_dependencies(list(final_plugins))
98 result.system_check_results = system_results
99
100 # Remove plugins with failed system dependencies
101 failed_plugins = []
102 for plugin in list(final_plugins):
103 plugin_metadata = PLUGIN_METADATA[plugin]
104 for sys_dep in plugin_metadata.system_dependencies:
105 if not system_results.get(sys_dep, False):
106 failed_plugins.append(plugin)
107 result.warnings.append(
108 f"Plugin '{plugin}' disabled: missing system dependency '{sys_dep}'"
109 )
110 break
111
112 final_plugins -= set(failed_plugins)
113 result.removed_plugins.extend(failed_plugins)
114
115 # Check for circular dependencies
116 circular_deps = self._detect_circular_dependencies(list(final_plugins))
117 if circular_deps:
118 result.warnings.append(f"Circular dependencies detected: {circular_deps}")
119
120 result.final_plugins = sorted(list(final_plugins))
121
122 # Set overall status
123 if result.errors:
124 result.status = ResolutionStatus.ERROR
125 elif result.warnings:
126 result.status = ResolutionStatus.WARNING
127
128 return result

130 def _validate_plugins(self, plugins: List[str]) -> Tuple[List[str], List[str]]:
131 """Separate valid and invalid plugin names."""
132 valid = [p for p in plugins if p in PLUGIN_METADATA]
133 invalid = [p for p in plugins if p not in PLUGIN_METADATA]
134 return valid, invalid

136 def _get_plugin_dependencies(self, plugin_name: str) -> List[str]:
137 """Get all dependencies for a plugin recursively."""
138 if plugin_name not in PLUGIN_METADATA:
139 return []
140
141 metadata = PLUGIN_METADATA[plugin_name]
142 dependencies = set(metadata.plugin_dependencies)
143
144 # Recursively add dependencies of dependencies
145 for dep in metadata.plugin_dependencies:
146 if dep in PLUGIN_METADATA:
147 sub_deps = self._get_plugin_dependencies(dep)
148 dependencies.update(sub_deps)
149
150 return list(dependencies)

152 def _detect_circular_dependencies(self, plugins: List[str]) -> List[str]:
153 """Detect circular dependencies in plugin list."""
154 # Simple cycle detection using DFS
155 visited = set()
156 rec_stack = set()
157
158 def has_cycle(plugin: str) -> bool:
159 if plugin in rec_stack:
160 return True
161 if plugin in visited:
162 return False
163
164 visited.add(plugin)
165 rec_stack.add(plugin)
166
167 if plugin in PLUGIN_METADATA:
168 for dep in PLUGIN_METADATA[plugin].plugin_dependencies:
169 if dep in plugins and has_cycle(dep):
170 return True
171
172 rec_stack.remove(plugin)
173 return False
174
175 circular = []
176 for plugin in plugins:
177 if plugin not in visited and has_cycle(plugin):
178 circular.append(plugin)
179
180 return circular

182 def _check_system_dependencies(self, plugins: List[str]) -> Dict[str, bool]:
183 """Check availability of system dependencies."""
184 all_deps = set()
185 for plugin in plugins:
186 if plugin in PLUGIN_METADATA:
187 all_deps.update(PLUGIN_METADATA[plugin].system_dependencies)
188
189 results = {}
190 for dep in all_deps:
191 results[dep] = self._check_system_dependency(dep)
192
193 return results

195 def _check_system_dependency(self, dependency: str) -> bool:
196 """Check if a specific system dependency is available."""
197 checkers = {
198 "cuda": self._check_cuda,
199 "optix": self._check_optix,
200 "opengl": self._check_opengl,
201 "glfw": self._check_glfw,
202 "glew": self._check_glew,
203 "freetype": self._check_freetype,
204 "imgui": self._check_imgui,
205 "x11": self._check_x11
206 }
207
208 checker = checkers.get(dependency)
209 if checker:
210 try:
211 return checker()
212 except Exception:
213 return False
214
215 # For unknown dependencies, assume available
216 return True

306 def validate_configuration(self, plugins: List[str]) -> Dict[str, any]:
307 """Comprehensive validation of plugin configuration."""
308 valid, invalid = self._validate_plugins(plugins)
309 platform_compatible = [p for p in valid if p in self.platform_plugins]
310 system_deps = self._check_system_dependencies(platform_compatible)
311

314 "invalid_plugins": invalid,

318 "gpu_required": any(PLUGIN_METADATA[p].gpu_required for p in platform_compatible),
319 "total_requested": len(plugins),
