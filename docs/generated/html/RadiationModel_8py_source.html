32 def __init__(self, context: Context):
33 """
34 Initialize RadiationModel with graceful plugin handling.
35
36 Args:
37 context: Helios Context instance
38
39 Raises:
40 RadiationModelError: If radiation plugin is not available
41 """
42 self.context = context
43 self.radiation_model = None
44
45 # Check plugin availability using registry
46 registry = get_plugin_registry()
47
48 if not registry.is_plugin_available('radiation'):
49 # Get helpful information about the missing plugin
50 plugin_info = registry.get_plugin_capabilities()
51 available_plugins = registry.get_available_plugins()
52
53 error_msg = (
54 "RadiationModel requires the 'radiation' plugin which is not available.\n\n"
55 "The radiation plugin provides GPU-accelerated ray tracing using OptiX.\n"
56 "System requirements:\n"
57 "- NVIDIA GPU with CUDA support\n"
58 "- CUDA Toolkit installed\n"
59 "- OptiX runtime (bundled with PyHelios)\n\n"
60 "To enable radiation modeling:\n"
61 "1. Build PyHelios with radiation plugin:\n"
62 " build_scripts/build_helios --plugins radiation\n"
63 "2. Or use a profile that includes radiation:\n"
64 " build_scripts/build_helios --profile gpu-accelerated\n"
65 f"\nCurrently available plugins: {available_plugins}"
66 )
67
68 # Suggest alternatives if available
69 alternatives = registry.suggest_alternatives('radiation')
70 if alternatives:
71 error_msg += f"\n\nAlternative plugins available: {alternatives}"
72 error_msg += "\nConsider using energybalance or leafoptics for thermal modeling."
73
74 raise RadiationModelError(error_msg)
75
76 # Plugin is available - create radiation model
77 try:
78 self.radiation_model = radiation_wrapper.createRadiationModel(context.getNativePtr())
79 if self.radiation_model is None:
80 raise RadiationModelError(
81 "Failed to create RadiationModel instance. "
82 "This may indicate a problem with the native library or GPU initialization."
83 )
84 logger.info("RadiationModel created successfully")
85
86 except Exception as e:
87 raise RadiationModelError(f"Failed to initialize RadiationModel: {e}")

98 logger.debug("RadiationModel destroyed successfully")
99 except Exception as e:
100 logger.warning(f"Error destroying RadiationModel: {e}")
