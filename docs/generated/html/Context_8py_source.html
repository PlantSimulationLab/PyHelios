28 def __post_init__(self):
29 """Calculate centroid from vertices if not provided."""
30 if self.centroid is None and self.vertices:
31 # Calculate centroid as average of vertices
32 total_x = sum(v.x for v in self.vertices)
33 total_y = sum(v.y for v in self.vertices)
34 total_z = sum(v.z for v in self.vertices)
35 count = len(self.vertices)
36 self.centroid = vec3(total_x / count, total_y / count, total_z / count)

120 def _validate_file_path(self, filename: str, expected_extensions: List[str] = None) -> str:
121 """Validate and normalize file path for security.
122
123 Args:
124 filename: File path to validate
125 expected_extensions: List of allowed file extensions (e.g., ['.ply', '.obj'])
126
127 Returns:
128 Normalized absolute path
129
130
131 Raises:
132 ValueError: If path is invalid or potentially dangerous
133 FileNotFoundError: If file does not exist
134 """
135 import os.path
136
137 # Convert to absolute path and normalize
138 abs_path = os.path.abspath(filename)
139
140 # Check for path traversal attempts
141 if '..' in filename or abs_path != os.path.normpath(abs_path):
142 raise ValueError(f"Invalid file path (potential path traversal): {filename}")
143
144 # Check if file exists
145 if not os.path.exists(abs_path):
146 raise FileNotFoundError(f"File not found: {abs_path}")
147
148 # Check if it's actually a file (not a directory)
149 if not os.path.isfile(abs_path):
150 raise ValueError(f"Path is not a file: {abs_path}")
151
152 # Check file extension if specified
153 if expected_extensions:
154 file_ext = os.path.splitext(abs_path)[1].lower()
155 if file_ext not in [ext.lower() for ext in expected_extensions]:
156 raise ValueError(f"Invalid file extension '{file_ext}'. Expected one of: {expected_extensions}")
157
158 return abs_path

415 def addTrianglesFromArrays(self, vertices: np.ndarray, faces: np.ndarray,
416 colors: Optional[np.ndarray] = None) -> List[int]:
417 """
418 Add triangles from NumPy arrays (compatible with trimesh, Open3D format).
419
420 Args:
421 vertices: NumPy array of shape (N, 3) containing vertex coordinates as float32/float64
422 faces: NumPy array of shape (M, 3) containing triangle vertex indices as int32/int64
423 colors: Optional NumPy array of shape (N, 3) or (M, 3) containing RGB colors as float32/float64
424 If shape (N, 3): per-vertex colors
425 If shape (M, 3): per-triangle colors
426
427 Returns:
428 List of UUIDs for the added triangles
429
430 Raises:
431 ValueError: If array dimensions are invalid
432 """
433 # Validate input arrays
434 if vertices.ndim != 2 or vertices.shape[1] != 3:
435 raise ValueError(f"Vertices array must have shape (N, 3), got {vertices.shape}")
436 if faces.ndim != 2 or faces.shape[1] != 3:
437 raise ValueError(f"Faces array must have shape (M, 3), got {faces.shape}")
438
439 # Check vertex indices are valid
440 max_vertex_index = np.max(faces)
441 if max_vertex_index >= vertices.shape[0]:
442 raise ValueError(f"Face indices reference vertex {max_vertex_index}, but only {vertices.shape[0]} vertices provided")
443
444 # Validate colors array if provided
445 per_vertex_colors = False
446 per_triangle_colors = False
447 if colors is not None:
448 if colors.ndim != 2 or colors.shape[1] != 3:
449 raise ValueError(f"Colors array must have shape (N, 3) or (M, 3), got {colors.shape}")
450 if colors.shape[0] == vertices.shape[0]:
451 per_vertex_colors = True
452 elif colors.shape[0] == faces.shape[0]:
453 per_triangle_colors = True
454 else:
455 raise ValueError(f"Colors array shape {colors.shape} doesn't match vertices ({vertices.shape[0]},) or faces ({faces.shape[0]},)")
456
457 # Convert arrays to appropriate data types
458 vertices_float = vertices.astype(np.float32)
459 faces_int = faces.astype(np.int32)
460 if colors is not None:
461 colors_float = colors.astype(np.float32)
462
463 # Add triangles
464 triangle_uuids = []
465 for i in range(faces.shape[0]):
466 # Get vertex indices for this triangle
467 v0_idx, v1_idx, v2_idx = faces_int[i]
468
469 # Get vertex coordinates
470 vertex0 = vertices_float[v0_idx].tolist()
471 vertex1 = vertices_float[v1_idx].tolist()
472 vertex2 = vertices_float[v2_idx].tolist()
473
474 # Add triangle with or without color
475 if colors is None:
476 # No color specified
477 uuid = context_wrapper.addTriangle(self.context, vertex0, vertex1, vertex2)
478 elif per_triangle_colors:
479 # Use per-triangle color
480 color = colors_float[i].tolist()
481 uuid = context_wrapper.addTriangleWithColor(self.context, vertex0, vertex1, vertex2, color)
482 elif per_vertex_colors:
483 # Average the per-vertex colors for the triangle
484 color = np.mean([colors_float[v0_idx], colors_float[v1_idx], colors_float[v2_idx]], axis=0).tolist()
485 uuid = context_wrapper.addTriangleWithColor(self.context, vertex0, vertex1, vertex2, color)
486
487 triangle_uuids.append(uuid)
488
489 return triangle_uuids

557 def setPrimitiveData(self, uuid: int, label: str, value) -> None:
558 """
559 Set primitive data for a specific primitive with automatic type detection.
560
561 Args:
562 uuid: UUID of the primitive
563 label: String key for the data
564 value: Value to store (int, uint, float, double, bool*, string, vec2, vec3, vec4, int2, int3, int4, etc.)
565 *bool values are stored as int (0/1) since Helios core doesn't support native bool
566 """
567 # Handle basic types
568 if isinstance(value, bool):
569 # Bool is not supported by Helios core - treat as int (0/1)
570 int_value = 1 if value else 0
571 context_wrapper.setPrimitiveDataInt(self.context, uuid, label, int_value)
572 elif isinstance(value, int):
573 # For now, treat all Python int as signed int. Could add uint detection later
574 if value < 0 or value <= 2147483647: # Check if it fits in signed int
575 context_wrapper.setPrimitiveDataInt(self.context, uuid, label, value)
576 else:
577 # Use uint for larger positive values
578 context_wrapper.setPrimitiveDataUInt(self.context, uuid, label, value)
579 elif isinstance(value, float):
580 # Could add double detection for high precision values
581 context_wrapper.setPrimitiveDataFloat(self.context, uuid, label, value)
582 elif isinstance(value, str):
583 context_wrapper.setPrimitiveDataString(self.context, uuid, label, value)
584
585 # Handle Helios vector types
586 elif isinstance(value, vec2):
587 context_wrapper.setPrimitiveDataVec2(self.context, uuid, label, value.x, value.y)
588 elif isinstance(value, vec3):
589 context_wrapper.setPrimitiveDataVec3(self.context, uuid, label, value.x, value.y, value.z)
590 elif isinstance(value, vec4):
591 context_wrapper.setPrimitiveDataVec4(self.context, uuid, label, value.x, value.y, value.z, value.w)
592 elif isinstance(value, int2):
593 context_wrapper.setPrimitiveDataInt2(self.context, uuid, label, value.x, value.y)
594 elif isinstance(value, int3):
595 context_wrapper.setPrimitiveDataInt3(self.context, uuid, label, value.x, value.y, value.z)
596 elif isinstance(value, int4):
597 context_wrapper.setPrimitiveDataInt4(self.context, uuid, label, value.x, value.y, value.z, value.w)
598
599 # Handle duck typing and list/tuple conversions
600 elif hasattr(value, 'x') and hasattr(value, 'y') and hasattr(value, 'z') and hasattr(value, 'w'):
601 # Handle any vec4-like object (duck typing)
602 context_wrapper.setPrimitiveDataVec4(self.context, uuid, label, float(value.x), float(value.y), float(value.z), float(value.w))
603 elif hasattr(value, 'x') and hasattr(value, 'y') and hasattr(value, 'z'):
604 # Handle any vec3-like object (duck typing)
605 context_wrapper.setPrimitiveDataVec3(self.context, uuid, label, float(value.x), float(value.y), float(value.z))
606 elif hasattr(value, 'x') and hasattr(value, 'y') and not hasattr(value, 'z'):
607 # Handle any vec2-like object (duck typing)
608 context_wrapper.setPrimitiveDataVec2(self.context, uuid, label, float(value.x), float(value.y))
609
610 # Handle lists and tuples
611 elif isinstance(value, (list, tuple)):
612 if len(value) == 2:
613 # Handle as vec2 or int2 based on element type
614 if all(isinstance(v, int) for v in value):
615 context_wrapper.setPrimitiveDataInt2(self.context, uuid, label, int(value[0]), int(value[1]))
616 else:
617 context_wrapper.setPrimitiveDataVec2(self.context, uuid, label, float(value[0]), float(value[1]))
618 elif len(value) == 3:
619 # Handle as vec3 or int3 based on element type
620 if all(isinstance(v, int) for v in value):
621 context_wrapper.setPrimitiveDataInt3(self.context, uuid, label, int(value[0]), int(value[1]), int(value[2]))
622 else:
623 context_wrapper.setPrimitiveDataVec3(self.context, uuid, label, float(value[0]), float(value[1]), float(value[2]))
624 elif len(value) == 4:
625 # Handle as vec4 or int4 based on element type
626 if all(isinstance(v, int) for v in value):
627 context_wrapper.setPrimitiveDataInt4(self.context, uuid, label, int(value[0]), int(value[1]), int(value[2]), int(value[3]))
628 else:
629 context_wrapper.setPrimitiveDataVec4(self.context, uuid, label, float(value[0]), float(value[1]), float(value[2]), float(value[3]))
630 else:
631 raise ValueError(f"Unsupported list/tuple length: {len(value)}. Supported lengths: 2, 3, 4")
632 else:
633 raise ValueError(f"Unsupported primitive data type: {type(value)}. "
634 f"Supported types: bool, int, float, str, vec2, vec3, vec4, int2, int3, int4, "
635 f"or 2/3/4-element list/tuple")

683 def getPrimitiveData(self, uuid: int, label: str, data_type: type):
684 """
685 Get primitive data for a specific primitive with explicit type specification.
686
687 Args:
688 uuid: UUID of the primitive
689 label: String key for the data
690 data_type: Python type to retrieve (int, uint, float, double, bool, str, vec2, vec3, vec4, int2, int3, int4, etc.)
691
692 Returns:
693 The stored value of the specified type
694 """
695 # Handle basic types
696 if data_type == int:
697 return context_wrapper.getPrimitiveDataInt(self.context, uuid, label)
698 elif data_type == float:
699 return context_wrapper.getPrimitiveDataFloat(self.context, uuid, label)
700 elif data_type == bool:
701 # Bool is not supported by Helios core - get as int and convert
702 int_value = context_wrapper.getPrimitiveDataInt(self.context, uuid, label)
703 return int_value != 0
704 elif data_type == str:
705 return context_wrapper.getPrimitiveDataString(self.context, uuid, label)
706
707 # Handle Helios vector types
708 elif data_type == vec2:
709 coords = context_wrapper.getPrimitiveDataVec2(self.context, uuid, label)
710 return vec2(coords[0], coords[1])
711 elif data_type == vec3:
712 coords = context_wrapper.getPrimitiveDataVec3(self.context, uuid, label)
713 return vec3(coords[0], coords[1], coords[2])
714 elif data_type == vec4:
715 coords = context_wrapper.getPrimitiveDataVec4(self.context, uuid, label)
716 return vec4(coords[0], coords[1], coords[2], coords[3])
717 elif data_type == int2:
718 coords = context_wrapper.getPrimitiveDataInt2(self.context, uuid, label)
719 return int2(coords[0], coords[1])
720 elif data_type == int3:
721 coords = context_wrapper.getPrimitiveDataInt3(self.context, uuid, label)
722 return int3(coords[0], coords[1], coords[2])
723 elif data_type == int4:
724 coords = context_wrapper.getPrimitiveDataInt4(self.context, uuid, label)
725 return int4(coords[0], coords[1], coords[2], coords[3])
726
727 # Handle extended numeric types (require explicit specification since Python doesn't have these as distinct types)
728 elif data_type == "uint":
729 return context_wrapper.getPrimitiveDataUInt(self.context, uuid, label)
730 elif data_type == "double":
731 return context_wrapper.getPrimitiveDataDouble(self.context, uuid, label)
732
733 # Handle list return types (for convenience)
734 elif data_type == list:
735 # Default to vec3 as list for backward compatibility
736 return context_wrapper.getPrimitiveDataVec3(self.context, uuid, label)
737 elif data_type == "list_vec2":
738 return context_wrapper.getPrimitiveDataVec2(self.context, uuid, label)
739 elif data_type == "list_vec4":
740 return context_wrapper.getPrimitiveDataVec4(self.context, uuid, label)
741 elif data_type == "list_int2":
742 return context_wrapper.getPrimitiveDataInt2(self.context, uuid, label)
743 elif data_type == "list_int3":
744 return context_wrapper.getPrimitiveDataInt3(self.context, uuid, label)
745 elif data_type == "list_int4":
746 return context_wrapper.getPrimitiveDataInt4(self.context, uuid, label)
747
748 else:
749 raise ValueError(f"Unsupported primitive data type: {data_type}. "
750 f"Supported types: int, float, bool, str, vec2, vec3, vec4, int2, int3, int4, "
751 f"'uint', 'double', list (for vec3), 'list_vec2', 'list_vec4', 'list_int2', 'list_int3', 'list_int4'")
